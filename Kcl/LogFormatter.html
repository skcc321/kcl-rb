def call(severity, time, _progname, msg)
  payload = extract(msg)

  hash = {
    pid: Process.pid,
    level: severity,
    time: format_datetime(time).strip,
    **uuid,
    **payload,
    **tags
  }

  "#{hash.to_json}\n"
end
